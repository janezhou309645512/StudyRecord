示例：查询多个表的满足条件的字段
select m.fnumber as fnumber,m.FQTY as fqty,material.fnumber as fproduct,m.FBaseStatus as status 
from T_MM_MANUFACTUREORDER m 
join T_MM_MFTOrderStock s on s.fparentid = m.fid 
join t_bd_material material 
on m.FMATERIALID = material.fid 
where m.fnumber='WHR180606017' and m.fstorageorgunitid='/OQAAADENwvM567U'
